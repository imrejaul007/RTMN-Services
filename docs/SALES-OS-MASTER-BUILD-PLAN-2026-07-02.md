```typescript
// src/index.ts
import { Router } from 'express';
import { v4 as uuidv4 } from 'uuid';

const router = Router();

// Full CustomerTwin interface
export interface CustomerTwin {
  id: string;
  customerId: string;
  
  // Identity
  identity: {
    name: string;
    email: string;
    phone: string;
    company: string;
    title?: string;
    industry: string;
    size: 'startup' | 'smb' | 'mid' | 'enterprise';
  };
  
  // Lifecycle
  lifecycle: {
    stage: 'prospect' | 'lead' | 'customer' | 'expanding' | 'churning' | 'churned';
    since: Date;
    tenure: number;
    healthScore: number;
    ltv: number;
  };
  
  // Relationships
  relationships: {
    champion?: { name: string; email: string; influence: number };
    economicBuyer?: { name: string; email: string };
    blockers: string[];
    influencers: string[];
  };
  
  // Financial
  financials: {
    arr: number;
    mrr: number;
    aov: number;
    orders: number;
    paymentStatus: 'current' | 'late' | 'overdue';
    cac?: number;
    margin?: number;
  };
  
  // Behavior
  behavior: {
    engagement: number;
    lastActive: Date;
    preferredChannel: 'whatsapp' | 'email' | 'call' | 'in_app';
    loginFrequency: number;
    featureAdoption: string[];
    nps?: number;
  };
  
  // Intelligence
  intelligence: {
    churnRisk: number;
    expansionProbability: number;
    nextBestAction: string;
    recommendedOffer?: string;
    intentSignals: IntentSignal[];
  };
  
  // Memory
  memory: {
    lastInteraction: Date;
    preferences: Record<string, any>;
    keyOutcomes: string[];
    openIssues: string[];
  };
  
  // Relationships
  journeyTwins: string[];
  opportunityTwins: string[];
  
  // Meta
  confidence: number;
  lastUpdated: Date;
  createdAt: Date;
}

// CRUD Operations
const customerTwins = new Map<string, CustomerTwin>();

router.get('/', (req, res) => {
  const twins = Array.from(customerTwins.values());
  res.json({ success: true, count: twins.length, twins });
});

router.get('/:id', (req, res) => {
  const twin = customerTwins.get(req.params.id);
  if (!twin) return res.status(404).json({ error: 'Twin not found' });
  res.json({ success: true, twin });
});

router.post('/', (req, res) => {
  const twin: CustomerTwin = {
    id: uuidv4(),
    customerId: req.body.customerId,
    identity: req.body.identity,
    lifecycle: { ...req.body.lifecycle, since: new Date() },
    relationships: req.body.relationships || { blockers: [], influencers: [] },
    financials: req.body.financials,
    behavior: req.body.behavior,
    intelligence: { churnRisk: 50, expansionProbability: 50, nextBestAction: '', intentSignals: [] },
    memory: { lastInteraction: new Date(), preferences: {}, keyOutcomes: [], openIssues: [] },
    journeyTwins: [],
    opportunityTwins: [],
    confidence: 100,
    lastUpdated: new Date(),
    createdAt: new Date(),
  };
  
  customerTwins.set(twin.id, twin);
  res.status(201).json({ success: true, twin });
});

router.put('/:id', (req, res) => {
  const twin = customerTwins.get(req.params.id);
  if (!twin) return res.status(404).json({ error: 'Twin not found' });
  
  const updated = { ...twin, ...req.body, lastUpdated: new Date() };
  customerTwins.set(twin.id, updated);
  res.json({ success: true, twin: updated });
});

// Intelligence
router.post('/:id/score', (req, res) => {
  const twin = customerTwins.get(req.params.id);
  if (!twin) return res.status(404).json({ error: 'Twin not found' });
  
  // Calculate scores
  twin.intelligence.churnRisk = calculateChurnRisk(twin);
  twin.intelligence.expansionProbability = calculateExpansionProbability(twin);
  twin.intelligence.nextBestAction = recommendNextAction(twin);
  twin.lifecycle.healthScore = calculateHealthScore(twin);
  twin.confidence = Math.min(100, twin.confidence + 2);
  twin.lastUpdated = new Date();
  
  customerTwins.set(twin.id, twin);
  res.json({ success: true, twin });
});

router.post('/:id/simulate', async (req, res) => {
  const twin = customerTwins.get(req.params.id);
  if (!twin) return res.status(404).json({ error: 'Twin not found' });
  
  const { action, parameters } = req.body;
  const simTwin = JSON.parse(JSON.stringify(twin));
  
  switch (action) {
    case 'churn':
      simTwin.lifecycle.stage = 'churning';
      simTwin.intelligence.churnRisk = parameters.probability || 0.8;
      simTwin.lifecycle.ltv *= (1 - (parameters.revenueImpact || 0.3));
      break;
    case 'expand':
      simTwin.lifecycle.stage = 'expanding';
      simTwin.intelligence.expansionProbability = parameters.probability || 0.8;
      simTwin.lifecycle.ltv *= (1 + (parameters.revenueImpact || 0.2));
      break;
    case 'upgrade':
      simTwin.financials.arr *= 1.5;
      simTwin.financials.mrr *= 1.5;
      simTwin.lifecycle.ltv *= 1.3;
      break;
  }
  
  res.json({
    success: true,
    original: twin,
    simulated: simTwin,
    impact: {
      ltvChange: simTwin.lifecycle.ltv - twin.lifecycle.ltv,
      ltvChangePercent: `${((simTwin.lifecycle.ltv - twin.lifecycle.ltv) / twin.lifecycle.ltv * 100).toFixed(1)}%`,
    }
  });
});

// Helper functions
function calculateChurnRisk(twin: CustomerTwin): number {
  let risk = 50;
  if (twin.behavior.engagement < 30) risk += 20;
  if (twin.lifecycle.tenure < 3) risk += 15;
  if (twin.financials.paymentStatus === 'overdue') risk += 25;
  if (twin.behavior.loginFrequency < 2) risk += 10;
  return Math.min(100, risk);
}

function calculateExpansionProbability(twin: CustomerTwin): number {
  let prob = 30;
  if (twin.lifecycle.stage === 'customer') prob += 20;
  if (twin.behavior.engagement > 70) prob += 15;
  if (twin.lifecycle.tenure > 12) prob += 10;
  if (twin.behavior.nps && twin.behavior.nps > 8) prob += 15;
  return Math.min(100, prob);
}

function calculateHealthScore(twin: CustomerTwin): number {
  return Math.round(
    (100 - twin.intelligence.churnRisk) * 0.4 +
    twin.behavior.engagement * 0.3 +
    twin.lifecycle.tenure * 2 * 0.1 +
    (twin.financials.paymentStatus === 'current' ? 20 : 0) * 0.2
  );
}

function recommendNextAction(twin: CustomerTwin): string {
  if (twin.intelligence.churnRisk > 70) return 'retention_campaign';
  if (twin.intelligence.expansionProbability > 70) return 'expansion_offer';
  if (twin.behavior.engagement < 50) return 'engagement_nudge';
  if (twin.lifecycle.stage === 'lead') return 'schedule_discovery';
  return 'maintain_relationship';
}

export default router;
```

```typescript
export interface JourneyTwin {

  // Identity
  name: string;
  type: 'acquisition' | 'onboarding' | 'engagement' | 'expansion' | 'retention' | 'recovery';
  industry: string;
  
  // Stages
  stages: JourneyStage[];
  
  // Goals
  goals: JourneyGoal[];
  
  // Progress
  progress: {
    overall: number;
    velocity: number;
    expectedCompletion: Date;
    currentStage: number;
  };
  
  // Participants
  participants: {
    customers: string[];
    agents: string[];
    humans: string[];
  };
  
  // Outcomes
  outcomes: {
    completed: string[];
    pending: string[];
    failed: string[];
  };
  
  // Intelligence
  intelligence: {
    successFactors: string[];
    failureFactors: string[];
    recommendedActions: string[];
    similarJourneys: string[];
  };
  
  // Metrics
  metrics: {
    avgCompletionTime: number;
    successRate: number;
    avgEngagement: number;
  };
  
  status: 'draft' | 'active' | 'paused' | 'completed';
  createdAt: Date;
  updatedAt: Date;
}

export interface JourneyStage {
  id: string;
  name: string;
  order: number;
  entryCriteria: string[];
  exitCriteria: string[];
  activities: JourneyActivity[];
  milestones: string[];
  duration: { min: number; max: number };
  successRate: number;
  aiAgents: string[];
}

export interface JourneyActivity {
  id: string;
  type: 'email' | 'whatsapp' | 'call' | 'in_app' | 'survey';
  name: string;
  template?: string;
  scheduledAt?: string;
}

export interface JourneyGoal {
  id: string;
  type: 'activation' | 'revenue' | 'adoption' | 'engagement';
  metric: string;
  target: number;
  current: number;
  deadline?: Date;
  status: 'on_track' | 'at_risk' | 'behind' | 'achieved';
}

const journeyTwins = new Map<string, JourneyTwin>();

router.get('/', (req, res) => {
  const twins = Array.from(journeyTwins.values());
  res.json({ success: true, count: twins.length, twins });
});

router.get('/:id', (req, res) => {
  const twin = journeyTwins.get(req.params.id);
  if (!twin) return res.status(404).json({ error: 'JourneyTwin not found' });
  res.json({ success: true, twin });
});

router.post('/', (req, res) => {
  const twin: JourneyTwin = {
    id: uuidv4(),
    journeyId: req.body.journeyId,
    name: req.body.name,
    type: req.body.type,
    industry: req.body.industry || 'general',
    stages: req.body.stages || [],
    goals: req.body.goals || [],
    progress: {
      overall: 0,
      velocity: 0,
      expectedCompletion: req.body.deadline || new Date(Date.now() + 90 * 24 * 60 * 60 * 1000),
      currentStage: 0,
    },
    participants: { customers: [], agents: [], humans: [] },
    outcomes: { completed: [], pending: [], failed: [] },
    intelligence: {
      successFactors: [],
      failureFactors: [],
      recommendedActions: [],
      similarJourneys: [],
    },
    metrics: { avgCompletionTime: 0, successRate: 0, avgEngagement: 0 },
    status: 'draft',
    createdAt: new Date(),
    updatedAt: new Date(),
  };
  
  journeyTwins.set(twin.id, twin);
  res.status(201).json({ success: true, twin });
});

router.put('/:id/progress', (req, res) => {
  const twin = journeyTwins.get(req.params.id);
  if (!twin) return res.status(404).json({ error: 'JourneyTwin not found' });
  
  const { stageIndex, completed, engagement } = req.body;
  
  if (stageIndex !== undefined) {
    twin.progress.currentStage = stageIndex;
  }
  
  if (completed !== undefined) {
    twin.outcomes.completed.push(completed);
  }
  
  if (engagement !== undefined) {
    twin.metrics.avgEngagement = engagement;
  }
  
  twin.progress.overall = calculateProgress(twin);
  twin.updatedAt = new Date();
  
  journeyTwins.set(twin.id, twin);
  res.json({ success: true, twin });
});

function calculateProgress(twin: JourneyTwin): number {
  if (twin.stages.length === 0) return 0;

  const outcomeProgress = (twin.outcomes.completed.length / 
    (twin.outcomes.completed.length + twin.outcomes.pending.length + 1)) * 100;
  return Math.round((stageProgress + outcomeProgress) / 2);
}
```

```typescript
function evaluateConditions(conditions: TriggerCondition[], payload: any): boolean {
  if (conditions.length === 0) return true;
  return conditions.every(c => {
    const value = getNestedValue(payload, c.field);

      case 'eq': return value === c.value;

      case 'contains': return String(value).includes(c.value);

    }
  });
```

```typescript
// src/index.ts
import { Router } from 'express';
import { v4 as uuidv4 } from 'uuid';

const router = Router();

interface Experiment {
  id: string;
  name: string;
  type: 'ab_test' | 'multivariate' | 'bandit';
  hypothesis: string;
  variants: Variant[];
  metrics: Metric[];
  status: 'draft' | 'running' | 'paused' | 'completed';
  startDate?: Date;
  endDate?: Date;
  results?: ExperimentResults;
  createdAt: Date;
}

interface Variant {
  id: string;
  name: string;
  traffic: number;
  conversions: number;
  impressions: number;
}

interface Metric {
  name: string;
  type: 'conversion' | 'engagement' | 'revenue';
  control: number;
  treatment: number;
  lift: number;
  confidence: number;
}

interface ExperimentResults {
  winner?: string;
  lift: number;
  confidence: number;
  revenueImpact?: number;
}

const experiments = new Map<string, Experiment>();

router.get('/', (req, res) => {
  res.json({ success: true, experiments: Array.from(experiments.values()) });
});

router.post('/', (req, res) => {
  const experiment: Experiment = {
    id: uuidv4(),
    name: req.body.name,
    type: req.body.type,
    hypothesis: req.body.hypothesis,
    variants: req.body.variants.map((v: any) => ({
      id: uuidv4(),
      name: v.name,
      traffic: v.traffic || 50,
      conversions: 0,
      impressions: 0,
    })),
    metrics: req.body.metrics || [],
    status: 'draft',
    createdAt: new Date(),
  };
  
  experiments.set(experiment.id, experiment);
  res.status(201).json({ success: true, experiment });
});

router.put('/:id/start', (req, res) => {
  const experiment = experiments.get(req.params.id);
  if (!experiment) return res.status(404).json({ error: 'Experiment not found' });
  
  experiment.status = 'running';
  experiment.startDate = new Date();
  experiments.set(experiment.id, experiment);
  
  res.json({ success: true, experiment });
});

router.post('/:id/results', (req, res) => {
  const experiment = experiments.get(req.params.id);
  if (!experiment) return res.status(404).json({ error: 'Experiment not found' });
  
  const { variantId, impressions, conversions } = req.body;
  const variant = experiment.variants.find(v => v.id === variantId);
  
  if (variant) {
    variant.impressions += impressions;
    variant.conversions += conversions;
  }
  
  // Calculate results
  const control = experiment.variants[0];
  const treatment = experiment.variants[1];
  
  if (control && treatment) {
    const controlRate = control.impressions > 0 ? control.conversions / control.impressions : 0;
    const treatmentRate = treatment.impressions > 0 ? treatment.conversions / treatment.impressions : 0;
    
    const lift = controlRate > 0 ? ((treatmentRate - controlRate) / controlRate) * 100 : 0;
    const n = control.impressions + treatment.impressions;
    const pooled = (control.conversions + treatment.conversions) / n;
    const se = Math.sqrt(pooled * (1 - pooled) * (1/control.impressions + 1/treatment.impressions));
    const z = se > 0 ? (treatmentRate - controlRate) / se : 0;
    const confidence = Math.min(99.9, (1 - 2 * (1 - normalCDF(Math.abs(z)))) * 100);
    
    experiment.results = {
      winner: confidence > 95 && lift > 0 ? treatment.id : undefined,
      lift: treatmentRate - controlRate,
      confidence,
      revenueImpact: lift * treatment.impressions * (req.body.aov || 1000),
    };
  }
  
  experiments.set(experiment.id, experiment);
  res.json({ success: true, experiment });
});

function normalCDF(x: number): number {
  const a1 =  0.254829592;
  const a2 = -0.284496736;
  const a3 =  1.421413741;
  const a4 = -1.453152027;
  const a5 =  1.061405429;
  const p  =  0.3275911;
  
  const sign = x < 0 ? -1 : 1;
  x = Math.abs(x) / Math.sqrt(2);
  
  const t = 1.0 / (1.0 + p * x);
  const y = 1.0 - (((((a5 * t + a4) * t) + a3) * t + a2) * t + a1) * t * Math.exp(-x * x);
  
  return 0.5 * (1.0 + sign * y);
}
```
