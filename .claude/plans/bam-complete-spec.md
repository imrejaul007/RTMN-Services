### The recommendation algorithm

```typescript
class BAMRecommender {
  recommend(blueprint: CompanyBlueprint): Asset[] {
    // 1. Extract capabilities from blueprint
    const capabilities = this.extractCapabilities(blueprint);
    
    // 2. Find assets that match capabilities
    const matches = BAM.search({
      capabilities,
      industry: blueprint.industry,
      region: blueprint.region,
      compliance: blueprint.compliance
    });
    
    // 3. Rank by relevance + ratings + compatibility
    const ranked = this.rank(matches, {
      relevance: 0.4,
      rating: 0.3,
      compatibility: 0.2,
      popularity: 0.1
    });
    
    // 4. Return top 10-15 recommendations
    return ranked.slice(0, 15);
  }
}
```
