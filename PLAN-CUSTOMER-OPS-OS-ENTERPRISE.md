interface CustomerTwin {
  // Identity
  id: string;
  clientId: string;
  projectId: string;
  
  // Identity Layer
  identity: {
    email?: string;
    phone?: string;
    whatsapp?: string;
    deviceId?: string;
    socialIds?: Record<string, string>;
  };
  
  // Orders Layer
  orders: {
    total: number;
    totalSpent: number;
    avgOrderValue: number;
    lastOrderDate: Date;
    pending: number;
    returned: number;
    cancelled: number;
  };
  
  // Payments Layer
  payments: {
    successful: { count: number; amount: number };
    failed: { count: number; amount: number };
    refunded: { count: number; amount: number };
    chargebacks: number;
    preferredMethods: string[];
  };
  
  // Support Layer
  support: {
    totalTickets: number;
    openTickets: number;
    resolvedTickets: number;
    avgResolutionTime: number;
    csatScore: number;
    escalatedCount: number;
    lastContact: Date;
  };
  
  // Engagement Layer
  engagement: {
    sessions: number;
    frequency: 'daily' | 'weekly' | 'monthly';
    lastSeen: Date;
    pushOptIn: boolean;
    emailOptIn: boolean;
    whatsappOptIn: boolean;
    referrals: number;
    reviews: number;
  };
  
  // Journey Layer
  journey: {
    source: string;                    // How they found us
    signupDate: Date;
    firstPurchase: Date;
    milestones: JourneyMilestone[];
    touchpoints: Touchpoint[];         // Ad → Website → Signup → Purchase → Support
  };
  
  // Memory Layer
  memory: {
    preferences: string[];
    facts: Record<string, any>;
    relationships: string[];
    extractedFromConversations: string[];
  };
  
  // AI Predictions
  ai: {
    genuinenessScore: number;
    churnRisk: 'low' | 'medium' | 'high' | 'critical';
    csatProbability: number;
    escalationRisk: number;
    lifetimeValue: number;
    nextBestAction: string;
    upsellProbability: number;
    refundProbability: number;
  };
  
  // Cross-Ecosystem (RTMN unique)
  ecosystem: {
    // If same person uses other RTMN services
    hojaiUserId?: string;
    rezCoins?: number;
    stayOwnBookings?: number;
    adBazaarCampaigns?: number;
    corpPerksBenefits?: string[];
  };
  
  // Digital Passport
  passport: {
    trustLevel: 'verified' | 'unverified' | 'suspended';
    verificationMethods: string[];
    riskFlags: string[];
    accountAge: number;
    verifiedAccounts: string[];
  };
}
