RTMN Chat Widget SDK

Add real-time customer support chat to any website in minutes

Quick Start

Add the chat widget to your website with just two lines of code:

<!-- Add the SDK -->
<script src="https://cdn.your-domain.com/chat-widget.js"></script>

<!-- Initialize with your config -->
<script>
  window.RTMNChatWidget.init({
    apiUrl: 'https://api.your-domain.com',
    businessId: 'your-business-id',
    position: 'bottom-right'
  });
</script>

Configuration

API Methods

// Initialize the widget
const widget = window.RTMNChatWidget.init(config);

// Listen to events
widget.on('message:receive', (msg) => console.log(msg));

// Send a quick reply
widget.sendQuickReply('Hello!');

// Open/close programmatically
widget.isWidgetOpen(); // boolean

// Destroy the widget
window.RTMNChatWidget.destroy();

Events

const widget = window.RTMNChatWidget.init(config);

widget.on('ready', () => console.log('Widget ready'));
widget.on('open', () => console.log('Widget opened'));
widget.on('close', () => console.log('Widget closed'));
widget.on('message:send', (msg) => console.log('Sent:', msg));
widget.on('message:receive', (msg) => console.log('Received:', msg));
widget.on('typing:start', ({ sender }) => console.log('Typing:', sender));
widget.on('agent:joined', (agent) => console.log('Agent joined:', agent));
widget.on('agent:left', (agent) => console.log('Agent left:', agent));
widget.on('transfer:request', (dept) => console.log('Transfer requested:', dept));
widget.on('error', (err) => console.error(err));
