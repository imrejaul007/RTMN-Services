## Quick Start

### Prerequisites
- Xcode 15.0+
- CocoaPods
- iOS 15.0+ deployment target

### Build Instructions

```bash
cd iOS

# Generate Xcode project (DONE)
xcodegen generate

# Install dependencies (DONE)
pod install

# Open in Xcode
open RAZOKeyboard.xcworkspace

# Build (in Xcode)
# Select RAZOKeyboardApp scheme and press ⌘+B
```

### Project Structure

```
iOS/
├── RAZOKeyboard.xcworkspace     # Open this in Xcode
├── RAZOKeyboard.xcodeproj       # Generated by XcodeGen
├── Podfile                      # CocoaPods dependencies
├── RAZOKeyboardApp/             # Host app
├── RAZOKeyboardExtension/       # Keyboard extension
└── RAZOCredentialProvider/     # Password autofill
```

### Enable Keyboard in iOS Settings

1. Open **Settings** > **General** > **Keyboard** > **Keyboards**
2. Tap **Add New Keyboard...**
3. Select **RAZO**
4. Enable **Allow Full Access** for voice input

### Manual Build (Command Line)

```bash
cd iOS

# Generate project
xcodegen generate

# Pod install
pod install

# Build
xcodebuild -workspace RAZOKeyboard.xcworkspace \
  -scheme RAZOKeyboardApp \
  -configuration Debug \
  -destination 'platform=iOS Simulator,name=iPhone 15' \
  build
```

## Keyboard Extension

```swift
// RazoKeyboardViewController.swift
import UIKit
import Speech

class RazoKeyboardViewController: UIInputViewController {
    
    private var sdk: RazoKeyboardSDK!
    private var voiceButton: UIButton!
    private var suggestionsView: SuggestionsView!
    
    override func viewDidLoad() {
        super.viewDidLoad()
        setupSDK()
        setupUI()
    }
    
    private func setupSDK() {
        sdk = RazoKeyboardSDK.Builder()
            .userId(getUserId())
            .build()
    }
    
    private func setupUI() {
        // Voice button
        voiceButton = UIButton(type: .system)
        voiceButton.setImage(UIImage(systemName: "mic.fill"), for: .normal)
        voiceButton.addTarget(self, action: #selector(startVoiceInput), for: .touchUpInside)
        
        // Suggestions view
        suggestionsView = SuggestionsView()
    }
    
    @objc func startVoiceInput() {
        let recognizer = SFSpeechRecognizer()
        let request = SFSpeechAudioBufferRecognitionRequest()
        
        let inputNode = audioEngine.inputNode
        let recordingFormat = inputNode.outputFormat(forBus: 0)
        
        inputNode.installTap(onBus: 0, bufferSize: 1024, format: recordingFormat) { buffer, _ in
            request.append(buffer)
        }
        
        recognizer.recognitionTask(with: request) { [weak self] result, error in
            if let text = result?.bestTranscription.formattedString {
                self?.processText(text)
            }
        }
        
        audioEngine.prepare()
        audioEngine.start()
    }
    
    private func processText(_ text: String) {
        Task {
            let cleaned = try? await sdk.cleanupText(text)
            let suggestions = try? await sdk.getSuggestions(text)
            
            await MainActor.run {
                self.showSuggestions(suggestions)
            }
        }
    }
}
```

## Voice Input

```swift
// VoiceInputManager.swift
class VoiceInputManager: NSObject {
    
    private let sdk: RazoKeyboardSDK
    private let speechRecognizer = SFSpeechRecognizer(locale: Locale(identifier: "en-IN"))
    private var recognitionRequest: SFSpeechAudioBufferRecognitionRequest?
    
    func startListening(completion: @escaping (String) -> Void) {
        SFSpeechRecognizer.requestAuthorization { status in
            guard status == .authorized else { return }
        }
        
        recognitionRequest = SFSpeechAudioBufferRecognitionRequest()
        
        speechRecognizer?.recognitionTask(with: recognitionRequest!) { result, error in
            if let text = result?.bestTranscription.formattedString {
                completion(text)
            }
        }
    }
    
    func stopListening() {
        recognitionRequest?.endAudio()
    }
}
```

class GrammarManager {

        }

## Passkeys (WebAuthn)

```swift
// PasskeyManager.swift
class PasskeyManager {
    
    private let sdk: RazoKeyboardSDK
    
    func createPasskey(for site: String) async throws -> String {
        let challenge = generateChallenge()
        
        let credential = try await ASAuthorizationPlatformKeyCredentialGenerator().createCredential(
            assertion: ASAuthorizationPublicKeyCredentialProvider(
                relyingPartyIdentifier: site
            ),
            challenge: challenge
        )
        
        try await sdk.createPasskey(site: site)
        
        return credential.credentialId
    }
    
    func authenticate(with passkeyId: String, for site: String) async throws -> Bool {
        return try await sdk.authenticateWithPasskey(site: site)
    }
}
```

// BiometricAuth.swift
class BiometricAuth {
    
    func authenticate(reason: String = "Unlock RAZO Vault") async -> Bool {
        let context = LAContext()
        var error: NSError?
        
        guard context.canEvaluatePolicy(.deviceOwnerAuthenticationWithBiometrics, error: &error) else {
            return false
        }
        
        return await withCheckedContinuation { continuation in
            context.evaluatePolicy(
                .deviceOwnerAuthenticationWithBiometrics,
                localizedReason: reason
            ) { success, _ in
                continuation.resume(returning: success)
            }
        }
    }
}

class SyncManager {
    
    private let sdk: RazoKeyboardSDK

        syncTimer = Timer.scheduledTimer(withTimeInterval: 30, repeats: true) { [weak self] _ in

                try? await self?.sdk.sync()
