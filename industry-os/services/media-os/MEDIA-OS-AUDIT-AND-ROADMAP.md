### How Media OS Consumes AdBazaar

```typescript
// AdBazaar Integration Module
class AdBazaarIntegration {
  
  // 1. DSP Integration (for buying ads)
  async buyAdCampaign(campaign: AdCampaign): Promise<CampaignResult> {
    const response = await fetch('http://localhost:4990/api/campaigns', {
      method: 'POST',
      headers: {
        'Authorization': `Bearer ${this.getToken()}`,
        'Content-Type': 'application/json'
      },
      body: JSON.stringify({
        advertiserId: campaign.advertiserId,
        name: campaign.name,
        targeting: {
          demographics: campaign.targetDemographics,
          contentCategories: campaign.contentCategories,
          viewerTwins: campaign.audienceSegments
        },
        budget: campaign.budget,
        bidding: campaign.biddingStrategy
      })
    });
    return response.json();
  }
  
  // 2. SSP Integration (for selling inventory)
  async getAdInventory(slot: AdSlot): Promise<Inventory> {
    const response = await fetch(`http://localhost:4980/api/inventory/available`, {
      method: 'POST',
      headers: { 'Authorization': `Bearer ${this.getToken()}` },
      body: JSON.stringify({
        channelId: slot.channelId,
        programId: slot.programId,
        startTime: slot.startTime,
        duration: slot.duration,
        viewerTwin: slot.viewerTwin
      })
    });
    return response.json();
  }
  
  // 3. Audience Segments (from AdBazaar)
  async getAudienceSegments(): Promise<AudienceSegment[]> {
    const response = await fetch('http://localhost:4805/api/segments', {
      headers: { 'Authorization': `Bearer ${this.getToken()}` }
    });
    return response.json();
  }
  
  // 4. Attribution Tracking
  async trackConversion(data: ConversionData): Promise<void> {
    await fetch('http://localhost:4950/api/track', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        viewerId: data.viewerId,
        campaignId: data.campaignId,
        eventType: data.eventType,
        timestamp: new Date().toISOString(),
        metadata: data.metadata
      })
    });
  }
  
  // 5. Yield Optimization
  async optimizeYield(channelId: string): Promise<YieldRecommendation> {
    const response = await fetch(`http://localhost:4990/api/yield/optimize`, {
      method: 'POST',
      body: JSON.stringify({ channelId })
    });
    return response.json();
  }
}
```
